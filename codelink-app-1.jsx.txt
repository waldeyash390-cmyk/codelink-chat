import React, { useState, useEffect, useRef, useCallback } from 'react';
import { Send, Paperclip, Copy, Check, LogOut, Users, X, Loader2 } from 'lucide-react';

const THEME = {
  ink: '#000000',
  panel: '#0A0A0A',
  bone: '#FFFFFF',
  mist: '#9A9A9A',
  mistDim: '#5C5C5C',
  danger: '#FF6B6B',
  border: 'rgba(255,255,255,0.16)',
  borderStrong: 'rgba(255,255,255,0.3)',
};

const AVATAR_SHADES = ['#FFFFFF', '#D9D9D9', '#BFBFBF', '#A6A6A6', '#8C8C8C', '#737373'];

function hashColor(name) {
  let h = 0;
  for (let i = 0; i < name.length; i++) h = (h * 31 + name.charCodeAt(i)) % AVATAR_SHADES.length;
  return AVATAR_SHADES[Math.abs(h)];
}

function genCode() {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
  let s = '';
  for (let i = 0; i < 6; i++) s += chars[Math.floor(Math.random() * chars.length)];
  return s;
}

function timeLabel(ts) {
  return new Date(ts).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
}

const MAX_MESSAGES = 150;
const PRESENCE_TTL = 15000;

export default function App() {
  const [screen, setScreen] = useState('join');
  const [nameInput, setNameInput] = useState('');
  const [codeInput, setCodeInput] = useState('');
  const [username, setUsername] = useState('');
  const [roomCode, setRoomCode] = useState('');
  const [messages, setMessages] = useState([]);
  const [imageCache, setImageCache] = useState({});
  const [members, setMembers] = useState([]);
  const [text, setText] = useState('');
  const [error, setError] = useState('');
  const [joining, setJoining] = useState(false);
  const [sending, setSending] = useState(false);
  const [copied, setCopied] = useState(false);
  const [uploadingImg, setUploadingImg] = useState(false);

  const scrollRef = useRef(null);
  const fileInputRef = useRef(null);
  const pollRef = useRef(null);
  const presenceRef = useRef(null);
  const knownImgKeys = useRef(new Set());

  const messagesKey = `room:${roomCode}:messages`;
  const presenceKey = `room:${roomCode}:presence`;

  const fetchMessages = useCallback(async () => {
    try {
      const res = await window.storage.get(messagesKey, true);
      const list = res ? JSON.parse(res.value) : [];
      setMessages(list);
      list.forEach(async (m) => {
        if (m.type === 'image' && !knownImgKeys.current.has(m.imgKey)) {
          knownImgKeys.current.add(m.imgKey);
          try {
            const img = await window.storage.get(m.imgKey, true);
            if (img) setImageCache((prev) => ({ ...prev, [m.imgKey]: img.value }));
          } catch (e) { /* image not ready yet */ }
        }
      });
    } catch (e) {
      setMessages([]);
    }
  }, [messagesKey]);

  const fetchPresence = useCallback(async () => {
    try {
      const res = await window.storage.get(presenceKey, true);
      const map = res ? JSON.parse(res.value) : {};
      const now = Date.now();
      const active = Object.entries(map).filter(([, ts]) => now - ts < PRESENCE_TTL).map(([n]) => n);
      setMembers(active);
    } catch (e) {
      setMembers([]);
    }
  }, [presenceKey]);

  const heartbeat = useCallback(async () => {
    try {
      let map = {};
      try {
        const res = await window.storage.get(presenceKey, true);
        map = res ? JSON.parse(res.value) : {};
      } catch (e) { /* no presence yet */ }
      map[username] = Date.now();
      const now = Date.now();
      Object.keys(map).forEach((k) => { if (now - map[k] > PRESENCE_TTL) delete map[k]; });
      await window.storage.set(presenceKey, JSON.stringify(map), true);
    } catch (e) { /* best effort */ }
  }, [presenceKey, username]);

  useEffect(() => {
    if (screen !== 'chat') return;
    fetchMessages();
    fetchPresence();
    heartbeat();
    pollRef.current = setInterval(() => { fetchMessages(); fetchPresence(); }, 2500);
    presenceRef.current = setInterval(heartbeat, 5000);
    return () => {
      clearInterval(pollRef.current);
      clearInterval(presenceRef.current);
    };
  }, [screen, fetchMessages, fetchPresence, heartbeat]);

  useEffect(() => {
    if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
  }, [messages, imageCache]);

  const handleJoin = async (createNew) => {
    setError('');
    const name = nameInput.trim();
    if (!name) { setError('Enter a display name'); return; }
    const code = createNew ? genCode() : codeInput.trim().toUpperCase();
    if (!createNew && !code) { setError('Enter a room code or start a new room'); return; }
    setJoining(true);
    setUsername(name);
    setRoomCode(code);
    setScreen('chat');
    setJoining(false);
  };

  const pushMessage = async (msg) => {
    try {
      let list = [];
      try {
        const res = await window.storage.get(messagesKey, true);
        list = res ? JSON.parse(res.value) : [];
      } catch (e) { /* room has no messages yet */ }
      list.push(msg);
      if (list.length > MAX_MESSAGES) list = list.slice(list.length - MAX_MESSAGES);
      await window.storage.set(messagesKey, JSON.stringify(list), true);
      setMessages(list);
    } catch (e) {
      setError('Could not send. Try again.');
    }
  };

  const handleSend = async () => {
    const t = text.trim();
    if (!t || sending) return;
    setSending(true);
    setText('');
    await pushMessage({
      id: `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
      type: 'text',
      sender: username,
      content: t,
      timestamp: Date.now(),
    });
    setSending(false);
  };

  const handleImagePick = async (e) => {
    const file = e.target.files?.[0];
    e.target.value = '';
    if (!file) return;
    if (!file.type.startsWith('image/')) { setError('Please choose an image file'); return; }
    if (file.size > 2 * 1024 * 1024) { setError('Image must be under 2MB'); return; }
    setUploadingImg(true);
    setError('');
    try {
      const dataUrl = await new Promise((resolve, reject) => {
        const r = new FileReader();
        r.onload = () => resolve(r.result);
        r.onerror = reject;
        r.readAsDataURL(file);
      });
      const id = `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
      const imgKey = `room:${roomCode}:img:${id}`;
      await window.storage.set(imgKey, dataUrl, true);
      knownImgKeys.current.add(imgKey);
      setImageCache((prev) => ({ ...prev, [imgKey]: dataUrl }));
      await pushMessage({ id, type: 'image', sender: username, imgKey, timestamp: Date.now() });
    } catch (e) {
      setError('Image upload failed. Try a smaller file.');
    } finally {
      setUploadingImg(false);
    }
  };

  const copyCode = () => {
    if (navigator.clipboard) navigator.clipboard.writeText(roomCode).catch(() => {});
    setCopied(true);
    setTimeout(() => setCopied(false), 1500);
  };

  const leaveRoom = () => {
    clearInterval(pollRef.current);
    clearInterval(presenceRef.current);
    setScreen('join');
    setMessages([]);
    setMembers([]);
    setRoomCode('');
    setCodeInput('');
    knownImgKeys.current = new Set();
  };

  if (screen === 'join') {
    return (
      <div style={{ background: THEME.ink, minHeight: '100vh', color: THEME.bone }} className="flex items-center justify-center p-6">
        <style>{customCss}</style>
        <div className="w-full max-w-sm">
          <div className="mb-1">
            <span className="font-display text-4xl tracking-tight">LWYW Linking</span>
          </div>
          <p className="text-base mb-8 leading-relaxed tracking-wide" style={{ color: THEME.mist }}>Real-time group messaging for teams on the move.</p>

          <label className="block text-xs uppercase tracking-wider mb-1.5" style={{ color: THEME.mistDim }}>Display name</label>
          <input
            value={nameInput}
            onChange={(e) => setNameInput(e.target.value)}
            placeholder="What should people call you?"
            maxLength={24}
            className="w-full rounded-md px-3 py-2.5 mb-4 outline-none text-sm"
            style={{ background: THEME.panel, border: `1px solid ${THEME.border}`, color: THEME.bone }}
          />

          <label className="block text-xs uppercase tracking-wider mb-1.5" style={{ color: THEME.mistDim }}>Room code</label>
          <input
            value={codeInput}
            onChange={(e) => setCodeInput(e.target.value.toUpperCase())}
            placeholder="e.g. K7QX2M"
            maxLength={8}
            className="w-full rounded-md px-3 py-2.5 mb-3 outline-none text-sm font-mono tracking-widest"
            style={{ background: THEME.panel, border: `1px solid ${THEME.border}`, color: THEME.bone }}
          />

          {error && <p className="text-sm mb-3" style={{ color: THEME.danger }}>{error}</p>}

          <button
            onClick={() => handleJoin(false)}
            disabled={joining}
            className="w-full rounded-md py-2.5 text-sm font-medium mb-2.5 disabled:opacity-60"
            style={{ background: THEME.bone, color: THEME.ink }}
          >
            {joining ? 'Joining…' : 'Join room'}
          </button>
          <button
            onClick={() => handleJoin(true)}
            disabled={joining}
            className="w-full rounded-md py-2.5 text-sm font-medium disabled:opacity-60"
            style={{ background: 'transparent', border: `1px solid ${THEME.border}`, color: THEME.mist }}
          >
            Start a new room
          </button>
          <p className="text-xs mt-6 text-center" style={{ color: THEME.mistDim }}>
            Messages are stored for anyone with the room code — not end-to-end encrypted.
          </p>
        </div>
      </div>
    );
  }

  return (
    <div style={{ background: THEME.ink, color: THEME.bone }} className="h-screen flex flex-col md:flex-row overflow-hidden">
      <style>{customCss}</style>

      <button onClick={copyCode} className="fixed top-4 right-4 flex items-center gap-1.5 px-3 py-1.5 rounded-full text-xs font-mono tracking-widest" style={{ zIndex: 50, background: THEME.panel, border: `1px solid ${THEME.border}`, color: THEME.bone }}>
        {roomCode} {copied ? <Check size={12} /> : <Copy size={12} />}
      </button>

      <div className="flex md:flex-col items-center md:items-stretch justify-between md:justify-start gap-3 px-4 py-3 md:w-56 md:py-5" style={{ background: THEME.panel, borderBottom: `1px solid ${THEME.border}` }}>
        <div className="flex md:flex-col gap-1">
          <div className="flex items-center gap-2">
            <span className="font-display text-sm tracking-tight">LWYW Linking</span>
          </div>
        </div>
        <div className="hidden md:flex items-center gap-1.5 mt-5 text-xs" style={{ color: THEME.mist }}>
          <Users size={13} /> {members.length} online
        </div>
        <div className="hidden md:flex flex-col gap-1.5 mt-2">
          {members.map((m) => (
            <div key={m} className="flex items-center gap-2 text-xs" style={{ color: THEME.mist }}>
              <span className="w-1.5 h-1.5 rounded-full" style={{ background: hashColor(m) }} />
              {m}{m === username ? ' (you)' : ''}
            </div>
          ))}
        </div>
        <button onClick={leaveRoom} className="md:mt-auto flex items-center gap-1.5 text-xs px-2.5 py-1.5 rounded-md" style={{ color: THEME.mist }}>
          <LogOut size={13} /> <span className="hidden md:inline">Leave room</span>
        </button>
      </div>

      <div className="flex-1 flex flex-col min-h-0">
        <div ref={scrollRef} className="flex-1 overflow-y-auto px-4 md:px-8 py-5 scrollbar-thin">
          {messages.length === 0 && (
            <div className="h-full flex flex-col items-center justify-center text-center" style={{ color: THEME.mistDim }}>
              <p className="text-sm">No messages yet.</p>
              <p className="text-xs mt-1">Share code <span className="font-mono" style={{ color: THEME.bone }}>{roomCode}</span> to bring others in.</p>
            </div>
          )}
          <div className="max-w-2xl mx-auto flex flex-col gap-0.5">
            {messages.map((m, i) => {
              const mine = m.sender === username;
              const prevSame = i > 0 && messages[i - 1].sender === m.sender;
              const color = hashColor(m.sender);
              return (
                <div key={m.id} className={`flex gap-2.5 ${mine ? 'flex-row-reverse' : ''} ${prevSame ? 'mt-1' : 'mt-4'}`}>
                  <div className="flex flex-col items-center w-6">
                    {!prevSame ? (
                      <div className="w-6 h-6 rounded-full flex items-center justify-center font-medium" style={{ background: THEME.ink, border: `1.5px solid ${color}`, color: THEME.bone, fontSize: '10px' }}>
                        {m.sender.slice(0, 1).toUpperCase()}
                      </div>
                    ) : (
                      <div className="thread-line" style={{ background: color, opacity: 0.35 }} />
                    )}
                  </div>
                  <div className={`flex flex-col ${mine ? 'items-end' : 'items-start'}`} style={{ maxWidth: '75%' }}>
                    {!prevSame && (
                      <span className="text-xs mb-1 px-0.5" style={{ color: THEME.mistDim }}>{mine ? 'You' : m.sender} · {timeLabel(m.timestamp)}</span>
                    )}
                    {m.type === 'text' ? (
                      <div className="px-3 py-2 rounded-2xl text-sm leading-relaxed break-words" style={{ background: mine ? THEME.bone : 'transparent', color: mine ? THEME.ink : THEME.bone, border: mine ? 'none' : `1px solid ${THEME.border}` }}>
                        {m.content}
                      </div>
                    ) : (
                      <div className="rounded-xl overflow-hidden" style={{ border: `1px solid ${THEME.border}`, maxWidth: 240 }}>
                        {imageCache[m.imgKey] ? (
                          <img src={imageCache[m.imgKey]} alt="shared" className="block w-full" />
                        ) : (
                          <div className="w-40 h-28 flex items-center justify-center" style={{ background: THEME.panel }}>
                            <Loader2 size={16} className="animate-spin" style={{ color: THEME.mist }} />
                          </div>
                        )}
                      </div>
                    )}
                  </div>
                </div>
              );
            })}
          </div>
        </div>

        {error && (
          <div className="px-4 md:px-8 pb-1 text-xs flex items-center gap-2 max-w-2xl mx-auto w-full" style={{ color: THEME.danger }}>
            {error} <button onClick={() => setError('')}><X size={12} /></button>
          </div>
        )}

        <div className="px-4 md:px-8 pb-4 pt-2 max-w-2xl mx-auto w-full">
          <div className="flex items-end gap-2 rounded-2xl px-2 py-1.5" style={{ background: THEME.panel, border: `1px solid ${THEME.border}` }}>
            <input ref={fileInputRef} type="file" accept="image/*" className="hidden" onChange={handleImagePick} />
            <button onClick={() => fileInputRef.current?.click()} disabled={uploadingImg} className="p-2 rounded-full shrink-0" style={{ color: THEME.mist }}>
              {uploadingImg ? <Loader2 size={18} className="animate-spin" /> : <Paperclip size={18} />}
            </button>
            <textarea
              value={text}
              onChange={(e) => setText(e.target.value)}
              onKeyDown={(e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); handleSend(); } }}
              placeholder="Message…"
              rows={1}
              className="flex-1 bg-transparent outline-none resize-none text-sm py-2 max-h-28"
              style={{ color: THEME.bone }}
            />
            <button onClick={handleSend} disabled={!text.trim() || sending} className="p-2 rounded-full shrink-0 disabled:opacity-40" style={{ color: THEME.bone }}>
              <Send size={18} />
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}

const customCss = `
  body { font-family: -apple-system, system-ui, 'Segoe UI', Roboto, Helvetica, Arial, sans-serif; }
  input, textarea, button { font-family: inherit; }
  .font-display { font-family: -apple-system, system-ui, 'Segoe UI', Roboto, Helvetica, Arial, sans-serif; font-weight: 800; }
  .font-mono { font-family: ui-monospace, SFMono-Regular, Menlo, Consolas, 'Courier New', monospace; }
  .thread-line { width: 2px; height: 100%; min-height: 18px; border-radius: 1px; }
  .scrollbar-thin::-webkit-scrollbar { width: 6px; }
  .scrollbar-thin::-webkit-scrollbar-thumb { background: rgba(255,255,255,0.16); border-radius: 3px; }
  textarea::placeholder, input::placeholder { color: #5C5C5C; }
`;
